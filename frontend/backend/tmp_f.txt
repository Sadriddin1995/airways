import { request } from './client'

export type Flight = {
  id: number
  flightNumber: string
  departureTime: string
  arrivalTime: string
  basePrice?: string
  originAirportId: number
  destinationAirportId: number
  companyId: number
  planeId: number
  computedPrice?: string;\n  classPricing?: { classId: number; price?: string; seatLimit?: number }[]\n}\n\nexport type SearchParams = {
  origin: string
  destination: string
  date: string
  cabin?: string
  adults?: number
}

export function searchFlights(params: SearchParams) {
  const qs = new URLSearchParams(params as any).toString()
  return request<Flight[]>(`/flights/search?${qs}`)
}

export function getFlight(id: number) {
  return request<Flight>(`/flights/${id}`)
}
